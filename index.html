def create_backup():
    """
    Cria um backup conectando-se diretamente ao banco de dados com PyMySQL.
    """
    timestamp = datetime.now().strftime("%Y-%m-%d_%H-%M-%S")
    backup_filename = f"backup_{DB_NAME}_{timestamp}.sql"
    
    print(f"[{datetime.now()}] Criando backup (m√©todo PyMySQL): {backup_filename}...")
    db = None
    try:
        db = pymysql.connect(host=DB_HOST, user=DB_USER, password=DB_PASSWORD, database=DB_NAME)
        cursor = db.cursor()

        with open(backup_filename, "w", encoding='utf-8') as backup_file:
            cursor.execute("SHOW TABLES")
            tables = cursor.fetchall()

            for table_row in tables:
                table_name = table_row[0]
                
                # 1. Adiciona o comando para apagar a tabela (DROP TABLE)
                backup_file.write(f"\n-- Backup para a tabela {table_name}\n")
                backup_file.write(f"DROP TABLE IF EXISTS `{table_name}`;\n")

                # 2. Pega e escreve o comando de cria√ß√£o da tabela (CREATE TABLE)
                cursor.execute(f"SHOW CREATE TABLE `{table_name}`")
                create_table_statement = cursor.fetchone()[1]
                backup_file.write(f"{create_table_statement};\n\n")

                # 3. Pega e escreve os dados (INSERT INTO)
                cursor.execute(f"SELECT * FROM `{table_name}`")
                rows = cursor.fetchall()
                for row in rows:
                    formatted_values = []
                    for value in row:
                        if value is None:
                            formatted_values.append("NULL")
                        elif isinstance(value, (int, float)):
                            formatted_values.append(str(value))
                        else:
                            # Escapa aspas simples
                            escaped_value = str(value).replace("'", "''")
                            formatted_values.append(f"'{escaped_value}'")
                    
                    values_string = ", ".join(formatted_values)
                    backup_file.write(f"INSERT INTO `{table_name}` VALUES ({values_string});\n")
            
        print(f"Backup '{backup_filename}' criado com sucesso.")
        return backup_filename

    except Exception as e:
        print(f"Ocorreu um erro ao gerar o backup: {e}")
        if os.path.exists(backup_filename):
            os.remove(backup_filename)
        return None
    finally:
        if db:
            db.close()
